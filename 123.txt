export const reduce = (func, acc, htmlList) => {
  const iter = (items, acc) => {
    if (isEmpty(items)) {
      return acc;
    }

    const item = head(items);
    const tailItems = tail(items);
    if (func(item, acc)) {
      return iter(tailItems, func(item, acc));
    }

    return iter(tailItems, acc);
  };
  return iter(htmlList, acc)
};
